1function G_zv = G_zv_gen(zv, m, q) 
 2% G_zv_gen - Function to generate G_z matrix given vector z(v)
 3%
 4% Syntax:
 5% G_zv = G_zv_gen(zv, m, q)
 6%
 7% Inputs:
 8%    zv   - NT x V, orignial imaging data matrix
 9%    m    - Number of Gaussian components in MoG
10%    q    - Number of Independent Components (IC)
11%
12% Outputs:
13%    G_zv   - G_z matrix of possible IC membership permutations
14%
15% See also: UpdateThetaBetaAprx_Vect.m, UpdateThetaBeta.m 
16
17    x=(1:q)';
18    x = double(x);
19    y=(x-1).*double(m) + zv;
20    G_zv = zeros (q, m*q);
21    G_zv( sub2ind(size(G_zv), x, y) ) = 1;
22end
